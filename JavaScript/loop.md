**FOR LOOP:**
FOR(LET I=1; I<=5; I++){
	CONSOLE.LOG(I);
}

**WHILE LOOP:**
LET I=1;
WHILE(I<=5){
	CONSOLE.LOG(I);
	I++;
}

**DO..WHILE LOOP**:
LET COUNT=1;
DO{
	CONSOLE.LOG(COUNT);
	COUNT++;
}WHILE(COUNT<=5);

**FOR..IN LOOP:**
LET ST={NAME="PHANI", AGE=19};
FOR(LET KEY IN ST){
	CONSOLE.LOG(KEY+":"+ST[KEY]);
}

**FOR..OF LOOP:**
LET FRUITS=["APPLE', "BANANA", "GRAPES"];
FOR(LET I OF FRUITS){
	CONSOLE.LOG(I);
}

*FUNCTION DECLARATION:-*

FUNCTION GREET(NAME){
	RETURN "HELLO," +NAME;
}
CONSOLE.LOG(GREET("JHON"));

*FUNCTION EXPRESSION:-*

LET GREET = FUNC(NAME){
	RETURN "HI" + NAME;
};
CONSOLE.LOG(GREET("ALICE"));
